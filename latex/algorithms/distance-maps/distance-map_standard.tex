\begin{algorithm}[!t]
\DontPrintSemicolon
\caption[Computation of the \emph{Standard} distance map
         ($\mathcal{M}_{1|2,\mathrm{dense}}$)]{%
         Computation of the \emph{Standard} distance map
         ($\mathcal{M}_{1|2,\mathrm{dense}}$).}
\label{alg:distance-maps:alignment:dmaps:dense}
\KwIn{Reference point cloud $\mathcal{P} = \{\mathbf{p}_i\}_{i=1}^{n}$,
Resized grid $\mathcal{M}_{1|2,\mathrm{dense}}$}
\KwOut{Dense data structure $\mathcal{M}_{1|2,\mathrm{dense}}$
(see \eqref{eq:distance-maps:alignment:dmaps:representations:level1} and
\eqref{eq:distance-maps:alignment:dmaps:representations:level2})}
$\mathcal{M}_{1|2,\mathrm{dense}}$.fill($\{-1,+\inf\}$)\tcp*{Initialization}
$\mathcal{Q}$.clear()\;
\For(\tcp*[f]{Seeding}){$i = 1$ \KwTo $n$}{
  \If{$\mathbf{p}_i$ is not a valid point}{
    \textbf{continue}\;
  }
  $(r, c) = \mathrm{global2local}(\mathbf{p}_i)$\;
  \If{$(r, c)$ is not inside the grid}{
    \textbf{continue}\;
  }
  $\mathcal{M}_{1|2,\mathrm{dense}}(r, c) = \{i,0\}$\;
  $\mathcal{Q}.\mathrm{push}\left(\&\mathcal{M}_{1|2,\mathrm{dense}}(r, c)\right)$\;
}
\While(\tcp*[f]{Wavefront propagation}){$\mathcal{Q}$ is not empty}{
  $q_k = \mathcal{Q}$.pop()\;
  $m = q_k$.cell()\;
  $k = m$.index()\;
  \If{$k < 0$}{\textbf{continue}}
  $(r_k,c_k) = \mathcal{M}_{1|2,\mathrm{dense}}$.getCellPos($\&m$)\;
  $(r_p,c_p) = \mathrm{global2local}(\mathbf{p}_{k})$\;
  \ForEach{$(r', c') \in \mathcal{N}_8(r_k,c_k)$}{
    $d_o = m$.dist()\;
    $d_k = \| (r', c')^T - (r_p,c_p)^T \|_2 \cdot \delta$\;
    \If(\tcp*[f]{Update if closer parent exists}){$(d_k < d_{\max}$ \textbf{or} $d_{\max} \leq 0)$ \textbf{and} $d_o > d_k$}{
      $\mathcal{M}_{1|2,\mathrm{dense}}(r', c') = \{k,d_k\}$\;
      $\mathcal{Q}.\mathrm{push}\left(\&\mathcal{M}_{1|2,\mathrm{dense}}(r', c')\right)$\;
    }
  }
}
\end{algorithm}
